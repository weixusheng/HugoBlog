void downadjust(int low, int high){
    int i = low, j = i*2;   //和两个子节点比较
    while(j <= high){
        if(j+1 <= high && heap[j+1] > heap[j]){
            j = j+1;    //挑选最大的子节点
        }
        if(heap[j] > heap[i]){
            wap(heap[j], heap[i]);
            i = j;
            j = i*2;    //交换,并递推处理交换的结点
        }
        else{
            break;
        }
    }
}

void upadjust(int low, int high){
    int i = high, j = i/2;  //和父节点比较
    while(j >= low){
        if(heap[j] < heap[i]){
            swap(heap[j], heap[i]);
            i = j;
            j = i/2;    //递推交换后的父节点
        }
        else{
            break;
        }
    }
}
