void union(int a, int b){
    int faa = findfather(a);    //寻找父节点
    int fab = findfather(b);
    if(faa != fab){
        father[faa] = fab;  //链接父节点
    }
}

int findfather(int x){
    int a = x;
    while(x != father[x]){
        x = father[x];
    }
    while(a != father[a]){
        int z = a;
        a = father[a];
        father[z] = x;
    }
    return x;
}

int findfather(int v){
    if(v == father[v]){
        return v;   //找到根节点
    }
    else{
        int f = findfather(father[v]);
        father[v] = f;  //赋值
        return f;   //返回根节点
    }
}
